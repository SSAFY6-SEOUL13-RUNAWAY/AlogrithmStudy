#include <iostream>

using namespace std;

int main() {

  int n, k;
  cin >> n >> k;

  int *coin = new int[n]();
  
  int i, result = 0;
  for(i = 0; i < n; i++){
    cin >> coin[i];
  }

  for(i = 0; i < n; i++){
    result += k / coin[n - i - 1];
    k = k % coin[n - i - 1];
    if(k == 0)
      break;
  }
  
  cout << result;

  return 0;
}
